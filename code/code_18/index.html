Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties: Integers in each row are sorted from left to right. The first integer of each row is greater than the last integer of the previous row. 

            
                
bool searchSmall(vector>& matrix, int target, int i, int j, int n, int m){
    if (i == n || j == m){
        return false;
    }
    if (matrix[i][j] == target) return true;
    if (matrix[n-1][j] < target || matrix[i][m-1] < target){
        if (matrix[n-1][j] < target) j++;
        if (matrix[i][m-1] < target) i++;
        return searchSmall(matrix, target, i, j, n, m);
    } else {
        for(int a = i; a < n; a++){
            for(int b = j; b < m; b++){
                if (matrix[a][b] == target) return true;
                if (matrix[a][b] > target) return false;
            }
        }
    }
    return false;            
}

bool searchMatrix(vector>& matrix, int target) {
    if (matrix.size()==0) return false;
