void swap(int *a, int *b){
    int temp = *a;
    *a = *b;
    *b = temp;
}

void bubble_sort(int *arr, size_t n){
    for(int i = 0 ; i < n - 1; i++){
        for(int j = 0; j < n - i - 1; j++){
            if (*(arr + j) > *(arr + j + 1)){
                swap((arr + j), (arr+ j + 1 ));
            }
        }
    }
}
