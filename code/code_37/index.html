The left subtree of root node contains node with key lesser than the root node’s key and the right subtree of root node contains node with key greater than the root node’s key. Hence, the tree is a BST.!!!code!!! 

            
                
!!!code!!!
bool isBST(Node* root) {
    if (root == NULL) return true;
    if (root->right != NULL)
        if (minVal(root->right) <= root->data) return false;
    if (root->left != NULL)
        if (maxVal(root->left) >= root->data) return false;
    return isBST(root->left) && isBST(root->right);
}
