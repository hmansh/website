Given a sorted array arr containing n elements with possibly duplicate elements, the task is to find indexes of first and last occurrences of an element x in the given array. 

            
                
int leftmost(int arr[], int low, int high, int x){
    if (low > high) return -1;
    int mid = low + (high - low)/2;
    if (arr[mid] == x) {
        if (mid == 0) return mid;
        if (arr[mid-1] == x){
            return leftmost(arr, low, mid-1, x);
        } else {
            return mid;
        }
    }
    if (arr[mid] > x){
        return leftmost(arr, low, mid-1, x);
    } 
    if (arr[mid] < x){
        return leftmost(arr, mid+1, high, x);
    }
}

int rightmost(int arr[], int low, int high, int x){
    if (low > high) return -1;
    int mid = low + (high - low)/2;
    if (arr[mid] == x) {
        if (mid == high) return mid;
        if (arr[mid+1] == x){
            return rightmost(arr, mid+1, high, x);
        } else {
            return mid;
        }
    }
    if (arr[mid] > x){
        return rightmost(arr, low, mid-1, x);
    } 
    if (arr[mid] < x){
        return rightmost(arr, mid+1, high, x);
    }
}

vector find(int arr[], int n , int x )
{
    return {leftmost(arr, 0, n, x),rightmost(arr, 0, n, x)};
}
