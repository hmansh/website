Given an array Arr of N positive integers. Your task is to find the elements whose value is equal to that of its index value. 

            
                
vector valueEqualToIndex(int arr[], int n) {
    vector res;
    for(int i = 0; i < n; i++){
        if (arr[i] == i + 1)
            res.push_back(i+1);
    }
    return res;
}
