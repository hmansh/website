Given a Binary Tree, print the diagonal traversal of the binary tree. Consider lines of slope -1 passing between nodes. Given a Binary Tree, print all diagonal elements in a binary tree belonging to same line. 

            
                
void traverseTree(Node* root, int d, map> &res){
    if (root == NULL) return;
    res[d].push_back(root->data);
    traverseTree(root->left, d + 1, res);
    traverseTree(root->right, d, res);
}


vector diagonal(Node *root){
    vector result;
    map> res;
    traverseTree(root, 0, res);
    for(auto i : res){
        vector v = i.second;
        for(int val : v){
           result.push_back(val);
        }
    }
    return result;
}
