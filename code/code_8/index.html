Given an array arr of N integers. Find the contiguous sub-array with maximum sum. 

            
                
int maxSubarraySum(int arr[], int n){
int currMax = 0;
int globalMax = 0;
for(int i = 0; i < n; i++){
    currMax = max(currMax + arr[i], arr[i]);
    if (currMax > globalMax){
        globalMax = currMax;
    }
}
return globalMax;
}
